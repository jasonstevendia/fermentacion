void setup() {
  lcd = new LiquidCrystal_I2C(0x27, 20, 4);
  lcd->init();
  lcd->backlight();
  lcd->setCursor(0,0);
  lcd->print("Bioelectronica");
  delay(3000);
  lcd->clear();
  Wire.begin();
  rtc.begin();
  if (!rtc.isrunning()) {
  rtc.adjust(DateTime(2021, 8, 5, 10, 26, 00)); // year, month, day, hour, min, sec
  }
  dht.begin();
  pinMode(A0, INPUT);
  pinMode(ledPin, OUTPUT); //interrupcion
  pinMode(ledPin2, OUTPUT); //interrupcion
  pinMode(ledPin3, OUTPUT); //interrupcion
  pinMode(ledPinA5, OUTPUT); //interrupcion
  attachInterrupt(D3, Stop, CHANGE);
  attachInterrupt(D2, Brix, CHANGE);
  attachInterrupt(D4, PH, CHANGE);
 pinMode(dsData ,INPUT);
 pinMode(D2 ,INPUT);
 pinMode(D3 ,INPUT);
 pinMode(D4 ,INPUT);
}

void OpB()
{   delay(500);
    lcd->clear();     
    
        
    monitorear=0;
  
    while(monitorear==0 )
    {
    digitalWrite(ledPin, HIGH);
    delay(2000);
    float _temp;
    int   i = 0;
    
    do {
    _temp = ds18b20.getTemperature();
  } while (!ds18b20.crcCheck() && MAXRETRY > i++);

  if (i < MAXRETRY) {
    celsius = _temp;
       // fahrenheit = ds18b20.convertToFahrenheit(_temp);
  // lcd->setCursor(10,1);
    
  }
  if(celsius==0) {
    celsius = NAN;
    lcd->setCursor(10,1);
    lcd->print("Failed ds18b20!");
    delay(300);
    lcd->setCursor(10,1);
    lcd->print("      ");
    
  }
   else    
       {
           lcd->setCursor(0,0); 
         lcd->print("Sensor ds18b20 Ok!");
         lcd->setCursor(0,1);
          lcd->print("T-Tanque:");
        lcd->setCursor(10,1);
        lcd->print(celsius);
        
        }
        
        
        
        
        delay(1000);
        float hmd = dht.readHumidity();   // Read humidity
        float temc = dht.readTemperature();   // Read temperature as Celsius doublecambie
        if (isnan(hmd) || isnan(temc) ) {
        lcd->setCursor(0,2);
        lcd->print("                    ");
        lcd->setCursor(0,2);
        lcd->print("Failed DHT sensor!");
        delay(500);
        lcd->setCursor(0,2);
        lcd->print("                  ");
             
         }
         else
         {
          lcd->setCursor(0,2);
        lcd->print("                    ");
        lcd->setCursor(0,2);
        lcd->print("DHTSensor OK!");
        Particle.publish("readings", String::format("{\"Hum(\%)\": %4.2f, \"Temp(°C)\": %4.2f", hmd, temc));
        lcd->setCursor(0,3);
        lcd->print("H.R:");
        lcd->setCursor(5,3);
        lcd->print(hmd,1);
        lcd->setCursor(10,3);
        lcd->print("T-A:");
        lcd->setCursor(15,3);
        lcd->print(temc,1);
          delay(500);}
  
        
  delay(1000);
  monitorear=1;
   lcd->clear();
  }
  
  }//FinopcionB
